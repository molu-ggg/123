    #include <iostream>
#include<queue>
using namespace std;
queue<pair<int ,int> >q;
int vis[1000010];


int bfs(int A,int B,int n){
    //10 2 7
    int x;
       q.push(make_pair(A,0));
        while(!q.empty()){
        x=q.front().first;
        int step=q.front().second;
        q.pop();
        if(x==B){
                cout<<step<<endl;
                return 0;
        }
        if(x-1>=0&&!vis[x-1]){
            q.push(make_pair(x-1,step+1));
            vis[x-1]=true;
        }
        if(x+1<=n&&!vis[x+1]){
            q.push(make_pair(x+1,step+1));
            vis[x+1]=true;
        }
        if(x*2<=n&&!vis[x*2]){
            q.push(make_pair(x*2,step+1));
        }

        }

    }
int main(){
    int n,A,B;
    cin>>n>>A>>B;
   bfs(A,B,n);
    return 0;
}

    台阶问题：¶

    一共有n个台阶，每次你只能走一步或者两步，一共有多少种方法到达？

    这个可以想两头：

    最后到达第n台阶:

    可以：F_{n}=F_{n-1}或者 F_{n}=F_{n-2}

    F_{n-1}=F_{n-2}+或者F_{n-1}=F_{n-3}

    台阶数 种类数
    1 1
    2 2
    3 1+2=3
    4 3+2=5

    可以递推出：F_{n}=F_{n-1}+F_{n-2}

    n条线最多划分平面的问题:（3点不共点）

    1 2
    2 4
    3 7
    4 11
    5 16

    每一次都是前一次的数加上前两次的差+1

    F_{n}=F_{n-1}+F_{n-1}-F_{n-2}+1

    就变成了F_{n}=2*F_{n-1}-F_{n-2}+1

    如果有时候：数很大的话，我们通常就是化时间为空间，降维，或者思考别的办法，特别是数学公式

    很显然，这里可以用数学公式： $$ F_{n}=\frac {n(n+1)}{2} $$

    错排问题：¶

    最后一步是： 前n-1封都排对了，只有最后一封没有排对，则将最后一封信跟前面的交换，容然成立，则种类数是：

    F_{n}=F_{n-1}*(n-1)

    这个时候要继续想：是否有其他情况的存在：

    前n-1封只有一封排对了，可以将该封与最后一封进行交换，现在想如何表示前n-1封中只有一封派对了那？

    继续研究前n-1封，假设前n-2封全错，只有最后一封对了，或者将某一位空余出来，其他的乱排，一共**n-1**种，则表示为是（n-1)*F_{n-2}

    则综合上述，表示为： $$ F_{n}=(n-1)*(F_{n-1}+F_{n-2}) $$

    #include<bits\stdc++.h>
using namespace std;
int vis[1000][1000];
int dx[]={1,1,-1,-1,2,2,-2,-2};
int dy[]={2,-2,2,-2,1,-1,1,-1};
long long int  F[1000][1000];
int main(){

     int n,m;
     int i,j;
     int hx,hy;
     cin>>n>>m>>hx>>hy;
     hx+=1;
     hy+=1;
     vis[hx][hy]=1;
     for(i=0;i<=m+2;i++){
        vis[0][i]=1;
        vis[n+2][i]=1;
     } 
     for(i=0;i<=n+2;i++){
        vis[i][0]=1;
        vis[i][m+2]=1;
     } 
     for(i=0;i<8;i++){
        if(hx+dx[i]<0||hy+dy[i]<0) continue;
        vis[hx+dx[i]][hy+dy[i]]=1;
     }
     F[1][1]=1;
     for(i=1;i<=n+1;i++){
        for(j=1;j<=m+1;j++){

            if(i==1&&j==1) continue;
            if(!vis[i][j])
            F[i][j]=F[i-1][j]+F[i][j-1];
        }

     }
     cout<<F[n+1][m+1];
}

    #include<bits\stdc++.h>
using namespace std;


int main(){


    int m,n,t,k,k1,k2;
    cin>>m>>n;
    k1=m,k2=n;
    if(m<n){
        t=m;m=n;n=t;
    }
    k=m%n;
    while(k) {
       m=n;
       n=k;
       k=m%n;   
    }
    cout<<n<<endl;//最大公因数
    cout<< (k1/n)*(k2/n)*n<<endl;

}
